            func: 'init',

                  instance: 'echo',

                  instance: 'echo',
                  func: 'echo'
                }],
                '/1/:a/2': [{
                  instance: 'echo',
                  func: 'echo'
                }],
                '/1/:a/:b': [{
                  instance: 'echo',
                  func: 'echo'
                }],
                '/abc/:c': [{
                  func: function () {
                    console.log('Router executing a function', arguments);
                  }
                }]
              }
            }]
